169 bool readData(Stream& input_stream) {
170 if (input_stream.available() == 0) return false;
171 if (!is_active) return false;
172
173 uint8_t frame[kFrameLength] = {0};
174
175 uint8_t b = 0;
176 bool header_seen = false;
177 while (true) {
178 if (!read_byte(input_stream, b)) {
179 return false;
180 }
181
182 if (header_seen) {
183 if (b == kVerLen) {
184 frame[0] = kHeader;
185 frame[1] = kVerLen;
186 break;
187 }
188 header_seen = (b == kHeader);
189 } else {
190 header_seen = (b == kHeader);
191 }
192 }
193
194 for (size_t i = 2; i < kFrameLength; ++i) {
195 if (!read_byte(input_stream, frame[i])) {
196 return false;
197 }
198 }
199
200 return parsePacket(frame, kFrameLength);
201 }

223 bool parsePacket(const uint8_t* packet, size_t len) {
224 if (packet == nullptr) {
225 return false;
226 }
227
228 if (len != kFrameLength) {
229 return false;
230 }
231
232 if (packet[0] != kHeader || packet[1] != kVerLen) {
233 return false;
234 }
235
236 constexpr size_t crcIndex = 46; // CRC byte index in the native frame
237 const uint8_t received_crc = packet[crcIndex];
238 const uint8_t calculated_crc = calculateCRC8(packet, crcIndex);
239 if (received_crc != calculated_crc) {
240 return false;
241 }
242
243 const uint16_t speed = readU16LE(packet + 2);
244 const uint16_t start_angle_raw = readU16LE(packet + 4);
245 const uint16_t end_angle_raw = readU16LE(packet + 42);
246 const uint16_t timestamp = readU16LE(packet + 44);
247
248 const float start_angle =
249 static_cast<float>(start_angle_raw % 36000U) / 100.0f;
250 const float end_angle = static_cast<float>(end_angle_raw % 36000U) / 100.0f;
251
252 float angle_diff = end_angle - start_angle;
253 while (angle_diff < 0.0f) {
254 angle_diff += 360.0f;
255 }
256 while (angle_diff >= 360.0f) {
257 angle_diff -= 360.0f;
258 }
259
260 const float angle_increment =
261 angle_diff / static_cast<float>(kPointsPerPacket - 1);
262
263 size_t offset = 6;
264 for (size_t i = 0; i < kPointsPerPacket; ++i) {
265 const uint16_t distance_mm = readU16LE(packet + offset);
266 const uint8_t intensity = packet[offset + 2];
267
268 float angle = start_angle + static_cast<float>(i) * angle_increment;
269 while (angle >= 360.0f) {
270 angle -= 360.0f;
271 }
272
273 const bool is_obstacle = (distance_mm > 0);
274 const float raw_distance_mm =
275 is_obstacle ? static_cast<float>(distance_mm)
276 : (max_range_m_ * 1000.0f);
277 const float converted_angle = convertAngle(angle);
278 const float converted_distance = convertDistance(raw_distance_mm);
279 if (on_point_ != nullptr) {
280 on_point_(
281 LidarResultData(converted_angle, converted_distance, intensity,
282 static_cast<uint64_t>(timestamp), is_obstacle),
283 ref);
284 } else {
285 // if no callback is registered, print the point data to Serial for
286 // debugging
287 char buffer[100];
288 snprintf(
289 buffer, sizeof(buffer),
290 "Angle: %.2f, Distance: %.2f, Intensity: %u, Timestamp: %u",
291 converted_angle, converted_distance, intensity, timestamp);
292 p_no_cb_output->println(buffer);
293 }
294 offset += 3;
295 }
296
297 out_speed_ = speed;
298 out_start_angle_ = convertAngle(start_angle);
299 out_end_angle_ = convertAngle(end_angle);
300 out_timestamp_ = timestamp;
301 return true;
302 }

304 float convertAngle(float angle_deg) const {
305 constexpr float kPi = 3.14159265358979323846f;
306 switch (angle_unit) {
307 case LidarAngleUnit::DEG:
308 return angle_deg;
309 case LidarAngleUnit::RAD:
310 return angle_deg * (kPi / 180.0f);
311 case LidarAngleUnit::DEG_ROS: {
312 return (angle_deg <= 180.0f) ? angle_deg : (angle_deg - 360.0f);
313 }
314 case LidarAngleUnit::RAD_ROS: {
315 const float angle_deg_ros =
316 (angle_deg <= 180.0f) ? angle_deg : (angle_deg - 360.0f);
317 return angle_deg_ros * (kPi / 180.0f);
318 }
319 default:
320 return angle_deg;
321 }
322 }
