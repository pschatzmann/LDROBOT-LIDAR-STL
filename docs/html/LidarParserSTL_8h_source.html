200 bool readData() {
201 if (read_byte_func == nullptr) return false;
202
203 if (!is_active) return false;
204
205 uint8_t frame[kFrameLength] = {0};
206
207 uint8_t b = 0;
208 bool header_seen = false;
209 while (true) {
210 if (!read_byte_func(b, read_byte_ref)) {
211 return false;
212 }
213
214 if (header_seen) {
215 if (b == kVerLen) {
216 frame[0] = kHeader;
217 frame[1] = kVerLen;
218 break;
219 }
220 header_seen = (b == kHeader);
221 } else {
222 header_seen = (b == kHeader);
223 }
224 }
225
226 for (size_t i = 2; i < kFrameLength; ++i) {
227 if (!read_byte_func(frame[i], read_byte_ref)) {
228 return false;
229 }
230 }
231
232 return parsePacket(frame, kFrameLength);
233 }

263 bool parsePacket(const uint8_t* packet, size_t len) {
264 if (packet == nullptr) {
265 return false;
266 }
267
268 if (len != kFrameLength) {
269 return false;
270 }
271
272 if (packet[0] != kHeader || packet[1] != kVerLen) {
273 return false;
274 }
275
276 constexpr size_t crcIndex = 46; // CRC byte index in the native frame
277 const uint8_t received_crc = packet[crcIndex];
278 const uint8_t calculated_crc = calculateCRC8(packet, crcIndex);
279 if (received_crc != calculated_crc) {
280 return false;
281 }
282
283 const uint16_t speed = readU16LE(packet + 2);
284 const uint16_t start_angle_raw = readU16LE(packet + 4);
285 const uint16_t end_angle_raw = readU16LE(packet + 42);
286 const uint16_t timestamp = readU16LE(packet + 44);
287
288 const float start_angle =
289 static_cast<float>(start_angle_raw % 36000U) / 100.0f;
290 const float end_angle = static_cast<float>(end_angle_raw % 36000U) / 100.0f;
291
292 float angle_diff = end_angle - start_angle;
293 while (angle_diff < 0.0f) {
294 angle_diff += 360.0f;
295 }
296 while (angle_diff >= 360.0f) {
297 angle_diff -= 360.0f;
298 }
299
300 const float angle_increment =
301 angle_diff / static_cast<float>(kPointsPerPacket - 1);
302
303 size_t offset = 6;
304 for (size_t i = 0; i < kPointsPerPacket; ++i) {
305 const uint16_t distance_mm = readU16LE(packet + offset);
306 const uint8_t intensity = packet[offset + 2];
307
308 float angle = start_angle + static_cast<float>(i) * angle_increment;
309 while (angle >= 360.0f) {
310 angle -= 360.0f;
311 }
312
313 const bool is_obstacle = (distance_mm > 0);
314 const float raw_distance_mm = is_obstacle
315 ? static_cast<float>(distance_mm)
316 : (max_range_m_ * 1000.0f);
317 const float converted_angle = convertAngle(angle);
318 const float converted_distance = convertDistance(raw_distance_mm);
319 if (on_point_ != nullptr) {
320 on_point_(
321 LidarResultData(converted_angle, converted_distance, intensity,
322 static_cast<uint64_t>(timestamp), is_obstacle),
323 ref);
324 } else {
325#ifdef ARDUINO
326 // if no callback is registered, print the point data to Serial for
327 // debugging
328 char buffer[100];
329 snprintf(buffer, sizeof(buffer),
330 "Angle: %.2f, Distance: %.2f, Intensity: %u, Timestamp: %u",
331 converted_angle, converted_distance, intensity, timestamp);
332 p_no_cb_output->println(buffer);
333#endif
334 }
335 offset += 3;
336 }
337
338 out_speed_ = speed;
339 out_start_angle_ = convertAngle(start_angle);
340 out_end_angle_ = convertAngle(end_angle);
341 out_timestamp_ = timestamp;
342 return true;
343 }

345 float convertAngle(float angle_deg) const {
346 constexpr float kPi = 3.14159265358979323846f;
347 switch (angle_unit) {
348 case LidarAngleUnit::DEG:
349 return angle_deg;
350 case LidarAngleUnit::RAD:
351 return angle_deg * (kPi / 180.0f);
352 case LidarAngleUnit::DEG_ROS: {
353 return (angle_deg <= 180.0f) ? angle_deg : (angle_deg - 360.0f);
354 }
355 case LidarAngleUnit::RAD_ROS: {
356 const float angle_deg_ros =
357 (angle_deg <= 180.0f) ? angle_deg : (angle_deg - 360.0f);
358 return angle_deg_ros * (kPi / 180.0f);
359 }
360 default:
361 return angle_deg;
362 }
363 }

367 case LidarDistanceUnit::MM:
368 return distance_mm;

371 case LidarDistanceUnit::M:

373 case LidarDistanceUnit::IN:
