234 bool readData() {
235 if (read_byte_func == nullptr) {
236 logger_.error("readData(): read byte callback is not set");
237 return false;
238 }
239
240 if (!is_active) {
241 logger_.warn("readData(): parser is not active, call begin() first");
242 return false;
243 }
244
245 uint8_t frame[kFrameLength] = {0};
246
247 uint8_t b = 0;
248 bool header_seen = false;
249 while (true) {
250 if (!read_byte_func(b, read_byte_ref)) {
251 logger_.warn("readData(): failed while waiting for frame header");
252 return false;
253 }
254
255 if (!header_seen) {
256 if (b == kHeader) {
257 header_seen = true;
258 }
259 continue;
260 }
261
262 if (b == kVerLen) {
263 frame[0] = kHeader;
264 frame[1] = kVerLen;
265 break;
266 }
267
268 // Keep sync if we received a repeated header byte, otherwise restart.
269 header_seen = (b == kHeader);
270 if (!header_seen) {
271 logger_.trace(
272 "readData(): header candidate rejected, second byte=0x%02X", b);
273 }
274 }
275
276 for (size_t i = 2; i < kFrameLength; ++i) {
277 if (!read_byte_func(frame[i], read_byte_ref)) {
278 logger_.warn("readData(): failed to read frame payload byte %u/%u",
279 static_cast<unsigned>(i + 1),
280 static_cast<unsigned>(kFrameLength));
281 return false;
282 }
283 }
284
285 return parsePacket(frame, kFrameLength);
286 }

308 bool parsePacket(const uint8_t* packet, size_t len) {
309 logger_.debug("parsePacket(): parsing packet with length=%u", static_cast<unsigned>(len));
310 if (packet == nullptr) {
311 logger_.error("parsePacket(): packet is null");
312 return false;
313 }
314
315 if (len != kFrameLength) {
316 logger_.error("parsePacket(): invalid frame length=%u expected=%u",
317 static_cast<unsigned>(len),
318 static_cast<unsigned>(kFrameLength));
319 return false;
320 }
321
322 if (packet[0] != kHeader || packet[1] != kVerLen) {
323 logger_.error(
324 "parsePacket(): invalid header bytes [0]=0x%02X [1]=0x%02X expected "
325 "0x%02X 0x%02X",
326 packet[0], packet[1], kHeader, kVerLen);
327 return false;
328 }
329
330 constexpr size_t crcIndex = 46; // CRC byte index in the native frame
331 const uint8_t received_crc = packet[crcIndex];
332 const uint8_t calculated_crc = calculateCRC8(packet, crcIndex);
333 if (received_crc != calculated_crc) {
334 logger_.error(
335 "parsePacket(): crc mismatch received=0x%02X calculated=0x%02X",
336 received_crc, calculated_crc);
337 return false;
338 }
339
340 const uint16_t speed = readU16LE(packet + 2);
341 const uint16_t start_angle_raw = readU16LE(packet + 4);
342 const uint16_t end_angle_raw = readU16LE(packet + 42);
343 const uint16_t timestamp = readU16LE(packet + 44);
344
345 const float start_angle =
346 static_cast<float>(start_angle_raw % 36000U) / 100.0f;
347 const float end_angle = static_cast<float>(end_angle_raw % 36000U) / 100.0f;
348
349 float angle_diff = end_angle - start_angle;
350 while (angle_diff < 0.0f) {
351 angle_diff += 360.0f;
352 }
353 while (angle_diff >= 360.0f) {
354 angle_diff -= 360.0f;
355 }
356
357 const float angle_increment =
358 angle_diff / static_cast<float>(kPointsPerPacket - 1);
359
360 size_t offset = 6;
361 for (size_t i = 0; i < kPointsPerPacket; ++i) {
362 const uint16_t distance_mm = readU16LE(packet + offset);
363 const uint8_t intensity = packet[offset + 2];
364
365 float angle = start_angle + static_cast<float>(i) * angle_increment;
366 while (angle >= 360.0f) {
367 angle -= 360.0f;
368 }
369
370 const bool is_obstacle = (distance_mm > 0);
371 const float raw_distance_mm = is_obstacle
372 ? static_cast<float>(distance_mm)
373 : (max_range_m_ * 1000.0f);
374 const float converted_angle = convertAngle(angle);
375 const float converted_distance = convertDistance(raw_distance_mm);
376 if (on_point_ != nullptr) {
377 on_point_(
378 LidarResultData(converted_angle, converted_distance, intensity,
379 static_cast<uint64_t>(timestamp), is_obstacle),
380 ref);
381 } else {
382 // if no callback is registered, print the point data to Serial for
383 // debugging
384 char buffer[100];
385 snprintf(buffer, sizeof(buffer),
386 "Angle: %.2f, Distance: %.2f, Intensity: %u, Timestamp: %u",
387 converted_angle, converted_distance, intensity, timestamp);
388 logger_.info(buffer);
389 }
390 offset += 3;
391 }
392
393 out_speed_ = speed;
394 out_start_angle_ = convertAngle(start_angle);
395 out_end_angle_ = convertAngle(end_angle);
396 out_timestamp_ = timestamp;
397 return true;
398 }

400 float convertAngle(float angle_deg) const {
401 constexpr float kPi = 3.14159265358979323846f;
402 switch (angle_unit) {
403 case LidarAngleUnit::DEG:
404 return angle_deg;
405 case LidarAngleUnit::RAD:
406 return angle_deg * (kPi / 180.0f);
407 case LidarAngleUnit::DEG_ROS: {
408 return (angle_deg <= 180.0f) ? angle_deg : (angle_deg - 360.0f);
409 }
410 case LidarAngleUnit::RAD_ROS: {
411 const float angle_deg_ros =
412 (angle_deg <= 180.0f) ? angle_deg : (angle_deg - 360.0f);
413 return angle_deg_ros * (kPi / 180.0f);
414 }
415 default:
416 return angle_deg;
417 }
418 }
