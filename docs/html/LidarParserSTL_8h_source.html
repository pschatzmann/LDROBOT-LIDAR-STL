240 bool readData() {
241 if (read_byte_func == nullptr) {
242 logger_.error("readData(): read byte callback is not set");
243 return false;
244 }
245
246 if (!is_active) {
247 logger_.warn("readData(): parser is not active, call begin() first");
248 return false;
249 }
250
251 uint8_t frame[kFrameLength] = {0};
252
253 uint8_t b = 0;
254 bool header_seen = false;
255 while (true) {
256 if (!read_byte_func(b, read_byte_ref)) {
257 logger_.warn("readData(): failed while waiting for frame header");
258 return false;
259 }
260
261 if (!header_seen) {
262 if (b == kHeader) {
263 header_seen = true;
264 }
265 continue;
266 }
267
268 if (b == kVerLen) {
269 frame[0] = kHeader;
270 frame[1] = kVerLen;
271 break;
272 }
273
274 // Keep sync if we received a repeated header byte, otherwise restart.
275 header_seen = (b == kHeader);
276 if (!header_seen) {
277 logger_.trace(
278 "readData(): header candidate rejected, second byte=0x%02X", b);
279 }
280 }
281
282 for (size_t i = 2; i < kFrameLength; ++i) {
283 if (!read_byte_func(frame[i], read_byte_ref)) {
284 logger_.warn("readData(): failed to read frame payload byte %u/%u",
285 static_cast<unsigned>(i + 1),
286 static_cast<unsigned>(kFrameLength));
287 return false;
288 }
289 }
290
291 return parsePacket(frame, kFrameLength);
292 }

314 bool parsePacket(const uint8_t* packet, size_t len) {
315 logger_.debug("parsePacket(): parsing packet with length=%u", static_cast<unsigned>(len));
316 if (packet == nullptr) {
317 logger_.error("parsePacket(): packet is null");
318 return false;
319 }
320
321 if (len != kFrameLength) {
322 logger_.error("parsePacket(): invalid frame length=%u expected=%u",
323 static_cast<unsigned>(len),
324 static_cast<unsigned>(kFrameLength));
325 return false;
326 }
327
328 if (packet[0] != kHeader || packet[1] != kVerLen) {
329 logger_.error(
330 "parsePacket(): invalid header bytes [0]=0x%02X [1]=0x%02X expected "
331 "0x%02X 0x%02X",
332 packet[0], packet[1], kHeader, kVerLen);
333 return false;
334 }
335
336 constexpr size_t crcIndex = 46; // CRC byte index in the native frame
337 const uint8_t received_crc = packet[crcIndex];
338 const uint8_t calculated_crc = calculateCRC8(packet, crcIndex);
339 if (received_crc != calculated_crc) {
340 logger_.error(
341 "parsePacket(): crc mismatch received=0x%02X calculated=0x%02X",
342 received_crc, calculated_crc);
343 return false;
344 }
345
346 const uint16_t speed = readU16LE(packet + 2);
347 const uint16_t start_angle_raw = readU16LE(packet + 4);
348 const uint16_t end_angle_raw = readU16LE(packet + 42);
349 const uint16_t timestamp = readU16LE(packet + 44);
350
351 const float start_angle =
352 static_cast<float>(start_angle_raw % 36000U) / 100.0f;
353 const float end_angle = static_cast<float>(end_angle_raw % 36000U) / 100.0f;
354
355 float angle_diff = end_angle - start_angle;
356 while (angle_diff < 0.0f) {
357 angle_diff += 360.0f;
358 }
359 while (angle_diff >= 360.0f) {
360 angle_diff -= 360.0f;
361 }
362
363 const float angle_increment =
364 angle_diff / static_cast<float>(kPointsPerPacket - 1);
365
366 size_t offset = 6;
367 for (size_t i = 0; i < kPointsPerPacket; ++i) {
368 const uint16_t distance_mm = readU16LE(packet + offset);
369 const uint8_t intensity = packet[offset + 2];
370
371 float angle = start_angle + static_cast<float>(i) * angle_increment;
372 while (angle >= 360.0f) {
373 angle -= 360.0f;
374 }
375
376 const bool is_obstacle = (distance_mm > 0);
377 const float raw_distance_mm = is_obstacle
378 ? static_cast<float>(distance_mm)
379 : (max_range_m_ * 1000.0f);
380 const float converted_angle = convertAngle(angle);
381 const float converted_distance = convertDistance(raw_distance_mm);
382 if (on_point_ != nullptr) {
383 on_point_(
384 LidarResultData(converted_angle, converted_distance, intensity,
385 static_cast<uint64_t>(timestamp), is_obstacle),
386 ref);
387 } else {
388 // if no callback is registered, print the point data to Serial for
389 // debugging
390 char buffer[100];
391 snprintf(buffer, sizeof(buffer),
392 "Angle: %.2f, Distance: %.2f, Intensity: %u, Timestamp: %u",
393 converted_angle, converted_distance, intensity, timestamp);
394 logger_.info(buffer);
395 }
396 offset += 3;
397 }
398
399 out_speed_ = speed;
400 out_start_angle_ = convertAngle(start_angle);
401 out_end_angle_ = convertAngle(end_angle);
402 out_timestamp_ = timestamp;
403 return true;
404 }

406 float convertAngle(float angle_deg) const {
407 constexpr float kPi = 3.14159265358979323846f;
408 switch (angle_unit) {
409 case LidarAngleUnit::DEG:
410 return angle_deg;
411 case LidarAngleUnit::RAD:
412 return angle_deg * (kPi / 180.0f);
413 case LidarAngleUnit::DEG_ROS: {
414 return (angle_deg <= 180.0f) ? angle_deg : (angle_deg - 360.0f);
415 }
416 case LidarAngleUnit::RAD_ROS: {
417 const float angle_deg_ros =
418 (angle_deg <= 180.0f) ? angle_deg : (angle_deg - 360.0f);
419 return angle_deg_ros * (kPi / 180.0f);
420 }
421 default:
422 return angle_deg;
423 }
424 }
